```cpp
class Solution {
private:

        int balance = 0;
        for (char c : s) {
            if (c == '(') balance++;
            else balance--;
            if (balance < 0) return false;
        }
        return balance == 0;
    }

        if (curr.length() == 2 * n) {
            if (isValid(curr)) res.push_back(curr);
            return;
        }
        generateAll(curr + '(', n, res);
        generateAll(curr + ')', n, res);
    }

public:
    vector<string> generateParenthesis(int n) {
        vector<string> res;
        generateAll("", n, res);
        return res;
    }
};
```

```cpp
        if (curr.length() == 2 * n) {

        vector<string> res;
        backtrack("", 0, 0, n, res);
        return res;
    }
```

1. Maintain counts of `open` and `close` parentheses.
2. Add `(` if `open < n`.
3. Add `)` only when `close < open` to ensure valid balance.
4. This ensures **only valid combinations** are formed.
